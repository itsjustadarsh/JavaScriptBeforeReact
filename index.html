Async, Await, Fetch

// Promises
console.log("Hello");
let name = "Pedro";
let firstName = "John";
let listOfMovies = fetch("url.com/movies");
console.log(listOfMovies); // you might have called it before fetching

const event = new Promise((resolve, reject) => {
    let name = "Pedro";
    name == "Pedro" ? resolve(name) : reject("Name was not Pedro");
})

event.then(name => {
    // in case we resolve will give us name
}).catch(err => {
    // in case of reject will get the err
}).finally(() => {
    // always be called no matter what happens
});

const axios = require("axios");
const data = axios.get("https://cat-fact.herokuapp.com/facts"); 
// npm install axios
data.then((res) => {
    console.log(res.data);
}).catch((err) => {
    console.log(err);
}).finally(() => {
    console.log("Promise was resolved successfully");
});

// Async Await is just a Syntactical Sugar
// removes the need for axios
// to await for something we need to have it inside the asynchronous functions
const fetchData = async () => {
    try {
        const data = await fetch("https://cat-fact.herokuapp.com/facts");
        console.log(data);
    } catch(err) {
        console.log(err);
    } finally {
        console.log("Hello");
    }
}
